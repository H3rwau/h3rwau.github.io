# 背包问题

## 01背包 每件物品最多使用一次

![](../../amWiki/images/DP.PNG)

**二维基本算法：**

```c++
#include <iostream>
#include <algorithm>

using namespace std;
const int N =1010;

int n,m;
int v[N],w[N];
int f[N][N];

int main()
{
    cin>>n>>m;
    for(int i=1;i<=n;i++)
        cin>>v[i]>>w[i];
    for(int i=1;j<=n;i++)
        for(int j=0;j<=m;j++)
        {
            f[i][j]=f[i-1][j];
            if(j>=v[i])f[i][j]=max(f[i][j],f[i-1][j-v[i]]+w[i])
        }
    cout<<f[n][m]<<endl;
    return 0;
}
```

**优化一维算法：**

```c++
#include <iostream>

        {
           f[j]=max(f[j],f[j-v[i]]+w[i])
        }

```



## 完全背包 每件物品有无限个

f[i,j]可以看成这个物品选k个

计算方法：

1.去掉k个物品i

2.求Max,f[i-1,j-k*v[i]]

3.再加回来k个物品i

f[i,j] = f[i-1,j-k*v[i]]+k\*w[i]

```c++
#include <iostream>
#include <algorithm>
using namespace std;
const int N =1010;

int n,m;
int v[N],w[N];
int f[N][N];

int main()
{
    cin>>n>>m;
    for(int i=1;i<=n;i++)cin>>v[i]>>w[i];
    for(int i=1;i<=n;i++)
        for(int j=0;j<=m;j++)
            for(int k=0;k*v[i]<=j;k++)
                f[i][j]=max(f[i][j],f[i-1][j-v[i]*k]+w[i]*k);
    cout<<f[n][m]<<endl;
    return 0;
}
```

## 多重背包 每件物品最多有s[i]个

```c++
#include <iostream>
#include <algorithm>
using namespace std;
const int N =1010;

int n,m;
int v[N],w[N],s[N];
int f[N][N];

int main()
{
    cin>>n>>m;
    for(int i=1;i<=n;i++)cin>>v[i]>>w[i]>>s[i];
    for(int i=1;i<=n;i++)
        for(int j=0;j<=m;j++)
            for(int k=0;k<=s[i]&&k*v[i]<=j;k++)
                f[i][j]=max(f[i][j],f[i-1][j-v[i]*k]+w[i]*k);
    cout<<f[n][m]<<endl;
    return 0;
}
```

**二进制优化方法：**

比如用1,2,4,8,...,512这十组数可以凑出来0~1023之间的任意数

假设s=200

1,2,4,8,16,32,64,73可以凑出0~200之间任意输数

1,2,4,8,16,32,64可以凑出0~127之间任意数

1,2,4,8，...,2^k,c （**c<2^(k+1**)）

1,2,4,8，...,2^k可以凑出0~2^(k+1)-1的任意数

在多重背包问题上

第i个物品有s个 ---> 拆分成logs组，对这些新拆分出来的进行一次01背包问题

原来复杂度是N\*V\*S,现在复杂度是N\*V\*logS

```c++
#include <iostream>
#include <algorithm>

using namespace std;
const int N =25000,M=2010;

int n,m;
int v[N],w[N];
int f[N];
int main(){
    cin>>n>>m;
    int cnt=0;
    for(int i=1;i<=n;i++)
    {
        int a,b,s;
        cin>>a>>b>>s;
        int k=1;
        while(k<=s)
        {
            cnt++;
            v[cnt]=a*k;
            w[cnt]=b*k;
            s-=k;
            k*=2;
        }
        if(s>0)
        {
            cnt++;
            v[cnt]=a*s;
            w[cnt]=b*s;
        }
    }
    n=cnt;  
    for(int i=1;i<=n;i++)
        for(int j=m;j>=v[i];j--)
            f[j]=max(f[j],f[j-v[i]]+w[i]);
   	cout<<f[m]<<endl;
    return 0;
}
```



## 分组背包 物品有n组，每组有若干个物品，每组只能选一个物品

```c++
#include <iostream>
#include <algorithm>

using namespace std;
const int N =110;
int n,m;
int v[N][N],w[N][N],s[N];
int f[N];

int main()
{
    cin>>n>>m;
    for(int i=1;i<=n;i++)
    {
        cin>>s[i];
        for(int j=0;j<s[i];j++)
            cin>>v[i][j]>>w[i][j];
    }
    for(int i=1;i<=n;i++)
        for(int j=m;j>=0;j--)
            for(int k=0;k<s[i];k++)
                if(v[i][k]<=j)
                    f[j]=max(f[j],f[j-v[i][k]]+w[i][k]);
    cout<<f[m]<<endl;
}
```



# 线性DP

## 最长公共子序列：

首先状态表示f[i,j] **所有在第一个序列的前i个字母中出现，且在第二个序列的前j个字母中出现的子序列**



f[i,j]可根据是否包含a[i]和b[j]分为四种情况：00,01,10,11

00表示f[i-1,j-1]

01的情况被包含在了f[i-1,j]中

10的情况被包含在了f[i,j-1]中

11的情况是f[i-1,j-1]+1

```c++
#include <iostream>
#include <algorithm>

using namespace std;
const int N =1010;
int n,m;
char a[N],b[N];
int f[N][N];

int main()
{
    scanf("%d%d",&n,&m);
    scanf("%s%s",a+1,b+1);

    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
        {
            f[i][j]=max(f[i-1][j],f[i][j-1]);
            if(a[i]==b[j]) f[i][j]=max(f[i][j],f[i-1][j-1]+1);
        }

    printf("%d\n",f[n][m]);
    return 0;
}
```



# 区间DP

## 合并石子

f[i,j]代表集合，所有将第i堆石子到第j堆石子合并成一堆石子的合并方式

属性是最小值min



状态表示 f[i,j]= min{f[i,k]+f[k+1,j]+s[j]-s[i-1]}  (s[i]代表前i堆石子的总重量，k=i~j-1)

```c++
#include <iostream>
#include <algorithm>

using namespace std;
const int N =310;

int n;
int s[N];
int f[N][N];

int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++) scanf("%d",&s[i]);

    for(int i=1;i<=n;i++)s[i]+=s[i-1];

    for(int len =2;len<=n;len++)
        for(int i=1;i+len-1<=n;i++)
        {
            int l=i,r=i+len-1;
            f[l][r]=1e8;
            for(int k=l;k<r;k++)
                f[l][r]=min(f[l][r],f[l][k]+f[k+1][r]+s[r]-s[l-1]);
        }
    printf("%d\n",f[1][n]);
    return 0;
}
```
