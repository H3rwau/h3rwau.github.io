# c++学习笔记——26.线程封装方法

1.Thread.hpp

```c++
#ifndef __WD_THREAD_H__
#define __WD_THREAD_H__

//不相关的头文件不要堆进来  ==> Keep your code clean
#include <pthread.h>

namespace wd
{

class Thread
{
public:
	Thread();

	void start();
	void join();

	virtual ~Thread();

private:
	static void * threadFunc(void*);
	virtual void run()=0;

private:
	pthread_t _pthid;
	bool _isRunning;
};

}


#endif

```

2.Thread.cc

```c++
//#include ....  //实现文件的第一行必须要是相应的头文件，不能是其他的头文件
#include "Thread.hpp"

namespace wd
{
	
Thread::Thread()
: _pthid(0)
, _isRunning(false)
{
}

Thread::~Thread()
{
	if(_isRunning) {
		pthread_detach(_pthid);	
		_isRunning = false;
	}
}

void Thread::start()
{
	pthread_create(&_pthid, nullptr, threadFunc, this);
	_isRunning = true;
}

//子线程执行函数(static)
void * Thread::threadFunc(void * arg)
{
	Thread * pthread = static_cast<Thread*>(arg);
	if(pthread) {
		pthread->run();
	}

	return nullptr;
}

void Thread::join()
{
	if(_isRunning) {
		pthread_join(_pthid, nullptr);//阻塞式函数
		_isRunning = false;
	}
}

}//end of namespace wd

```

3.TestThread.cc

```c++

#include "Thread.hpp"   //自定义头文件

#include <unistd.h>
#include <stdio.h>    //C头文件
#include <stdlib.h>
#include <time.h>

#include <iostream>		//C++头文件
#include <memory>

//#include <log4cpp/Category.hh> //第三方库头文件

using std::cout;
using std::endl;

class MyThread
: public wd::Thread
{
	void run()
	{
		::srand(::time(nullptr));
		int cnt = 20;
		while(cnt-- > 0) {
			cout << "sub thread " << pthread_self() 
				 << ": number = " << (::rand() % 100) << endl; 
			::sleep(1);
		}
	}
};

void test0()
{
	std::unique_ptr<wd::Thread> thread(new MyThread());
	thread->start();
	thread->join();
}
 
int main(void)
{
	test0();
	return 0;
}
```
