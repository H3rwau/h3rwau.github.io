```c++
#include <iostream>

void test()
{
    //对于文件输入流而言，默认情况下，当文件不存在则读失败
    ifstream ifs("Point.cc", std::ios::in | std::ios::ate);
    if(!ifs.good())
    {
        cerr << "ifstream is broken" << endl;
        return;
    }
    cout << "pos = " << ifs.tellg() << endl;

    //对于文件输出流而言，当文件不存在就创建文件
    //当文件存在的时候，清空原文件，然后继续写入
    ofstream ofs("wuhan.txt");
    if(!ofs.good())
    {
        cerr << "ofstream is broken" <<endl;
        ifs.close();
        return;
    }

    string word;
    while(getline(ifs, word))//输入流而言，默认以空格为分隔符
    {
        ofs << word << endl;
    }

    ifs.close();
    ofs.close();
}
```

```c++
    printVectorCapacity(vec);
    vec.push_back(5);

    vec.push_back(6);
    printVectorCapacity(vec);
    vec.push_back(4);
    printVectorCapacity(vec);
    vec.push_back(5);
    printVectorCapacity(vec);
    vec.push_back(6);

    //vector的遍历
```

### 3.explicit防止隐式转换

```c++
#include <iostream>

using std::cout;
using std::endl;

class Point
{
public:
    /* explicit //防止隐式转换*/
    Point(int ix = 0, int iy = 0)
    : _ix(ix)
    , _iy(iy)
    {
        cout << "Point(int = 0, int = 0)" << endl;
    }

    void print() const
    {
        cout << "(" << _ix
             << ", " << _iy
             << ")" << endl;
    }

    ~Point()
    {
        cout << "~Point()" << endl;
    }
private:
    int _ix;
    int _iy;
};

void test()
{
    Point pt1(1, 9);
    cout << "pt1 = ";
    pt1.print();

    cout << endl;
    //int ===> Point
    Point pt2 = 5; //Point(5, 0),隐式转换
    cout << "pt2 = ";
    pt2.print();
}
int main(int argc, char **argv)
{
    test();

    return 0;
}
```

### 4.字符串流

```c++
#include <iostream>
#include <sstream>
#include <string>
#include <fstream>

using std::cout;
using std::cerr;
using std::endl;
using std::ostringstream;
using std::istringstream;
using std::stringstream;
using std::string;
using std::ifstream;

void test1()
{
    //int===>string
    int number = 10;
    ostringstream oss;
    oss << number;
    string str1 = oss.str();
    cout << "str1 = " << str1 << endl;
}

void test2()
{
    int number1 = 10;
    int number2 = 20;
    stringstream ss;
    ss << "number1= " << number1 << " ,number2= " << number2 << endl;//进行格式转换
    string str2 = ss.str();
    cout << str2;

    string key;
    int value;
    while(ss >> key >> value, !ss.eof())//对于输入流，默认以空格为分隔符
    {
        cout << key << " ---> " << value << endl;
    }
}

void readConfig(const string &filename)
{
    ifstream ifs(filename);
    if(!ifs)
    {
        cerr << "open file " << filename << " error" << endl;
        return;
    }

    string line;
    while(getline(ifs, line))
    {
        string key, value;
        istringstream iss(line);
        iss >> key >> value;
        cout << key << " ---> " << value << endl;
    }
    ifs.close();
}
void test3()
{
    readConfig("my.conf");
}

int main(int argc, char **argv)
{
    test3();
    return 0;
}
```
